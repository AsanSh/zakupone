# 2. Проверьте что вы там
pwd
ls -la

# 4. Создайте .htaccess
cat > .htaccess << 'HTACCESSEND'
DirectoryIndex index.html
Options -Indexes +FollowSymLinks
HTACCESSEND

# 5. Установите права
chmod 644 index.html
chmod 644 .htaccess

# 6. Проверьте что файлы созданы
ls -la
